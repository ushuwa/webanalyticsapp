<script>
    /* --- JAVASCRIPT LOGIC --- */

    // State
    let currentStep = 0;
    let answers = {
        roof: null,     // { points: 0 }
        children: null, // { points: 0 }
        tv: null        // { points: 0 }
    };

    // DOM Elements
    const sections = document.querySelectorAll('.section');
    const navBtns = document.querySelectorAll('.nav-btn');
    const calcBtn = document.getElementById('calc-btn');

    // Navigation
    function setStep(stepIndex) {
        currentStep = stepIndex;
        
        // Update Sections
        sections.forEach((sec, idx) => {
            if (idx === stepIndex) sec.classList.add('active');
            else sec.classList.remove('active');
        });

        // Update Nav Buttons
        navBtns.forEach((btn, idx) => {
            if (idx === stepIndex) btn.classList.add('active');
            else btn.classList.remove('active');
        });
    }

    // Selection Logic
    function selectAnswer(category, idPart, points) {
        // Update State
        answers[category] = { points: points };

        // Visual Selection
        // Deselect all in this group
        const groupBtns = document.querySelectorAll(`[id^="${getPrefix(category)}"]`);
        groupBtns.forEach(btn => btn.classList.remove('selected'));

        // Select clicked
        const clickedBtn = document.getElementById(getPrefix(category) + '-' + idPart);
        if(clickedBtn) clickedBtn.classList.add('selected');

        checkSurveyComplete();
    }

    // Helper to map category to ID prefix
    function getPrefix(cat) {
        if (cat === 'roof') return 'roof';
        if (cat === 'children') return 'child';
        if (cat === 'tv') return 'tv';
        return '';
    }

    function checkSurveyComplete() {
        if (answers.roof && answers.children && answers.tv) {
            calcBtn.disabled = false;
        } else {
            calcBtn.disabled = true;
        }
    }

    function calculateAndShow() {
        // Calculate
        const roofPts = answers.roof.points;
        const childPts = answers.children.points;
        const tvPts = answers.tv.points;
        const total = roofPts + childPts + tvPts;

        // Update Display Text
        document.getElementById('score-roof').innerText = `+${roofPts} pts`;
        document.getElementById('score-children').innerText = `+${childPts} pts`;
        document.getElementById('score-tv').innerText = `+${tvPts} pts`;
        document.getElementById('score-total-val').innerText = total;

        // Lookup Table Logic
        let probability = 0;
        
        // Reset rows
        document.querySelectorAll('.lookup-table tr').forEach(r => r.classList.remove('highlight-row'));

        if (total < 10) {
            probability = 85;
            document.getElementById('row-very-high').classList.add('highlight-row');
        } else if (total < 20) {
            probability = 60;
            document.getElementById('row-high').classList.add('highlight-row');
        } else if (total < 30) {
            probability = 35;
            document.getElementById('row-mod').classList.add('highlight-row');
        } else {
            probability = 10;
            document.getElementById('row-low').classList.add('highlight-row');
        }

        document.getElementById('prob-display').innerText = `${probability}%`;

        // Move to next step
        setStep(2);
    }

    function resetApp() {
        answers = { roof: null, children: null, tv: null };
        document.querySelectorAll('.option-btn').forEach(b => b.classList.remove('selected'));
        checkSurveyComplete();
        setStep(0);
    }

</script>
